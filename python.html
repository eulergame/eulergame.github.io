install

  1. sudo apt update
  2. sudo apt install software-properties-common
  3. sudo add-apt-repository ppa:deadsnakes/ppa
  4. sudo apt install python3.7
  5. python3.7 --version

build from source

  1. sudo apt update
  2. sudo apt install build-essential zlib1g-dev libncurses5-dev libgdbm-dev libnss3-dev libssl-dev libreadline-dev libffi-dev wget
  3. wget https://www.python.org/ftp/python/3.7.3/Python-3.7.4.tar.xz
  4. tar -xf Python-3.7.4.tar.xz
  5. cd Python-3.7.4
  6. ./configure --enable-optimizations
  7. make -j 8
  8. sudo make altinstall
  9. python3.7 --version
